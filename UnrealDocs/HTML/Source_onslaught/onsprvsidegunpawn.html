class ONSPRVSideGunPawn extends ONSWeaponPawn;

function bool StopWeaponFiring()
{
	if ( bWeaponIsAltFiring && (Bot(Controller) != None) )
	{
		Gun.CalcWeaponFire();
		Gun.AltFire(Controller);
	}
	return Super.StopWeaponFiring();
}
