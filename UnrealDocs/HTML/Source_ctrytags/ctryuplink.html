class CtryUplink extends TcpLink;

var PlayerController ToResolve;
var bool bSucceeded,bNowARIN,bNowAPNIC;
var ClientHandler Handler;

function StartPacking()
{
	local IpAddr IP;
	local int i;
	
	BindPort();
	i = Handler.MasterHandler.GetResolvData(0);
	if( i==0 )
		Resolve("whois.ripe.net");
	else
	{
		IP.Addr = i;
		Resolved(IP);
	}
}
event Resolved( IpAddr Addr )
{
	if( !bNowARIN )
		Handler.MasterHandler.SetResolvData(Addr.Addr,0);
	else if( !bNowAPNIC )
		Handler.MasterHandler.SetResolvData(Addr.Addr,1);
	else Handler.MasterHandler.SetResolvData(Addr.Addr,2);
	Addr.Port = 43;
	Open(Addr);
}
event Opened()
{
	if( Handler==None || Handler.bDeleteMe )
	{
		bSucceeded = True;
		Close();
		Destroy();
		Return;
	}
	SendText(Handler.ClientIPAddress$Chr(10));
}
event ReceivedText( string Text )
{
	local int i;
	local string S;
	
	if( bNowAPNIC && (InStr(Text,"This network range is not allocated to APNIC.")>0 || InStr(Text,"Not allocated by APNIC")>0) )
	{
		Handler.CountryResolved("US");
		bSucceeded = True;
		Close();
		Return;
	}
	S = Text;
	i = InStr(Caps(Text),"COUNTRY:");
	if( i==-1 )
		Return; // Wait for possibly next packet.
	Text = Mid(Text,i+8);
	While( Left(Text,1)==" " )
		Text = Mid(Text,1);
	i = InStr(Text,Chr(10));
	Text = Left(Text,i);
	if( Len(Text)>4 || (Text~="AU" && !bNowAPNIC) ) // Not from EU
	{
		/*Handler.CountryResolved(ToResolve,"US");
		bSucceeded = True;
		Close();*/
		Return;
	}
	Handler.CountryResolved(Text);
	bSucceeded = True;
	Close();
}
event Closed()
{
	local IpAddr IP;
	local int i;
	
	if( bSucceeded || (bNowARIN && bNowAPNIC) )
	{
		if( !bSucceeded )
			Handler.CountryResolved("US");
		Destroy();
	}
	else if( !bNowARIN )
	{
		BindPort();
		bNowARIN = True;
		i = Handler.MasterHandler.GetResolvData(1);
		if( i==0 )
			Resolve("whois.arin.net");
		else
		{
			IP.Addr = i;
			Resolved(IP);
		}
	}
	else
	{
		BindPort();
		bNowAPNIC = True;
		i = Handler.MasterHandler.GetResolvData(2);
		if( i==0 )
			Resolve("whois.apnic.net");
		else
		{
			IP.Addr = i;
			Resolved(IP);
		}
	}
}
event ResolveFailed()
{
	Destroy();
}

defaultproperties
{
}
